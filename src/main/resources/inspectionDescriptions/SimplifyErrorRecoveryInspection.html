    val uio: UIO[Int]   = UIO(1)

    uio.option                      => uio.map(Some(_))
    uio.either                      => uio.map(Right(_))
    uio.orElseEither(task)          => uio.map(Left(_))
    uio <+> task                    => uio.map(Left(_))
